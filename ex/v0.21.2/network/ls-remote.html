Entry point for this command

 
int ls_remote(git_repository *repo, int argc, char **argv)
{
  int error;

  if (argc < 2) {
    fprintf(stderr, "usage: %s ls-remote <remote>\n", argv[-1]);
    return EXIT_FAILURE;
  }

  error = use_remote(repo, argv[1]);

  return error;
}
