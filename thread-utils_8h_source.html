00028 /*
00029  * How TLS works is compiler+platform dependant
00030  * Sources: http://en.wikipedia.org/wiki/Thread-Specific_Storage
00031  *          http://predef.sourceforge.net/precomp.html
00032  */
00033 
00034 #define GIT_HAS_TLS 1
00035 
00036 #if defined(__APPLE__) && defined(__MACH__)
00037 # undef GIT_TLS
00038 
00039 #elif defined(__GNUC__) || \
00040       defined(__SUNPRO_C) || \
00041       defined(__SUNPRO_CC) || \
00042       defined(__xlc__) || \
00043       defined(__xlC__)
00044 # define GIT_TLS __thread
00045 
00046 #elif defined(__INTEL_COMPILER)
00047 # if defined(_WIN32) || defined(_WIN32_CE)
00048 #  define GIT_TLS __declspec(thread)
00049 # else
00050 #  define GIT_TLS __thread
00051 # endif
00052 
00053 #elif defined(_WIN32) || \
00054       defined(_WIN32_CE) || \
00055       defined(__BORLANDC__)
00056 # define GIT_TLS __declspec(thread)
00057 
00058 #else
00059 # undef GIT_HAS_TLS
00060 # define GIT_TLS /* nothing: tls vars are thread-global */
00061 #endif
00062 
00063 /* sparse and cygwin don't grok thread-local variables */
00064 #if defined(__CHECKER__) || defined(__CYGWIN__)
00065 # undef GIT_HAS_TLS
00066 # undef GIT_TLS
00067 # define GIT_TLS
00068 #endif
00069 
00070 #endif /* INCLUDE_git_thread_utils_h__ */
